# create function to calculate category for conversions -
def category_conversion(conversion_col):
    category=[]
    for conversion in df[conversion_col]:
        if conversion < 6:
            category.append('Less than 6')
        elif 6<= conversion < 12:
            category.append('6-11')
        elif 12<= conversion < 16:
            category.append('12-15')
        else:
            category.append('More than 16')
    return category

df['Facebook Conversion Category']= category_conversion('Facebook Ad Conversions')
df['AdWords Conversion Category']= category_conversion('AdWords Ad Conversions')

Hypothesis Testing¶

Hypothesis: Facebook Ads greater number of conversion compared to AdWords Ads¶

Null Hypothesis: There is no differnce conversion between Facebook and AdWords¶

Facebook<=AdWords¶

Alternate Hypothesis: There is a differnce conversion between Facebook and AdWords¶

Facebook>AdWords¶

print('Mean Conversion \n')
print('Facebook:',round(df['Facebook Ad Conversions'].mean(),2))
print('AdWords:',round(df['AdWords Ad Conversions'].mean(),2))

t_stats, p_value = st.ttest_ind(a=df['Facebook Ad Conversions'],b=df['AdWords Ad Conversions'],equal_var=False)
print('TStatistic: ',t_stats,'\nP_value: ',p_value)

# significance value is 5%
if p_value <0.05:
    print('Reject Null Hypothesis')
else:
    print('Accept the Null Hypothesis')
